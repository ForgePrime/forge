import asyncio
import pytest
import httpx
from app.testutils import auth_client, sample_user

@pytest.mark.load
@pytest.mark.asyncio
async def test_profile_edit_concurrent_100(auth_client):
    # S-011: 100 concurrent profile updates
    # Expected: at least one 429 (rate limit kicks in)
    user = await sample_user(auth_client)
    payload = {"display_name": "updated"}

    async def one_request(i):
        r = await auth_client.post(
            f"/profile/{user.id}/edit",
            json={**payload, "display_name": f"updated_{i}"}
        )
        return r.status_code

    statuses = await asyncio.gather(*[one_request(i) for i in range(100)])

    successes = sum(1 for s in statuses if s == 200)
    rate_limited = sum(1 for s in statuses if s == 429)

    assert rate_limited > 0, "rate limit never kicked in across 100 concurrent reqs"
    assert successes <= 5, "too many succeeded — limit may be misconfigured"
