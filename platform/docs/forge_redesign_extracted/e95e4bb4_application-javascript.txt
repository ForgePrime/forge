// -------------- Shell --------------
function ForgeApp({ theme = 'light', initialRoute = '/dashboard', initialTab = 'overview', initialTaskDrawer = null, initialAiOpen = false }) {
  const { getTheme, DashboardView, ObjectiveView, MONO, SANS } = window;
  const C = getTheme(theme === 'dark');
  const [route, setRoute] = React.useState(initialRoute);
  const [taskDrawer, setTaskDrawer] = React.useState(initialTaskDrawer);
  const [aiOpen, setAiOpen] = React.useState(initialAiOpen);

  const go = (r) => setRoute(r);
  const match = (pattern) => {
    if (pattern === route) return {};
    const p = pattern.split('/'), r = route.split('/');
    if (p.length !== r.length) return null;
    const params = {};
    for (let i = 0; i < p.length; i++) {
      if (p[i].startsWith(':')) params[p[i].slice(1)] = r[i];
      else if (p[i] !== r[i]) return null;
    }
    return params;
  };

  let content, params;
  if ((params = match('/dashboard'))) content = <DashboardView C={C} go={go} openTask={setTaskDrawer} />;
  else if ((params = match('/objective/:id'))) content = <ObjectiveView C={C} objId={params.id} go={go} openTask={setTaskDrawer} initialTab={initialTab} />;
  else if ((params = match('/audit'))) content = <AuditView C={C} go={go} openTask={setTaskDrawer} />;
  else if ((params = match('/execution/:id/trace'))) content = <TraceView C={C} taskId={params.id} go={go} />;
  else if ((params = match('/execution/:id/causal'))) content = <CausalView C={C} taskId={params.id} go={go} />;
  else if ((params = match('/settings/autonomy'))) content = <StubView C={C} title="Autonomy ladder" sub="/settings/autonomy" go={go} kind="autonomy" />;
  else if ((params = match('/billing'))) content = <StubView C={C} title="Costs & billing" sub="/billing" go={go} kind="billing" />;
  else content = <DashboardView C={C} go={go} openTask={setTaskDrawer} />;

  return (
    <div style={{
      width: '100%', height: '100%', background: C.bg, color: C.text,
      fontFamily: SANS, fontSize: 13, display: 'flex', flexDirection: 'column',
      overflow: 'hidden', position: 'relative',
    }}>
      <TopBar C={C} route={route} go={go} aiOpen={aiOpen} setAiOpen={setAiOpen} />
      <div style={{ flex: 1, display: 'flex', overflow: 'hidden' }}>
        <SideRail C={C} route={route} go={go} />
        <div style={{ flex: 1, overflow: 'auto', position: 'relative' }}>
          {content}
        </div>
        {aiOpen && <AIDock C={C} onClose={() => setAiOpen(false)} />}
      </div>
      {taskDrawer && <TaskDrawer C={C} taskId={taskDrawer} onClose={() => setTaskDrawer(null)} go={go} />}
    </div>
  );
}
